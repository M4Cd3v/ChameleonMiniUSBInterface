15       /** 
16        * Initialize the Chameleon USB library. 
17        * (!!) THIS MUST BE CALLED (!!) before performing any operations with this 
18        * interface, or for most of the operations in the ChameleonDeviceConfig class. 
19        * @param mainActivity : a reference to the main Android app's activity (a simple "this" 
20        *                       reference should do it if called from within an activity method). 
21        * @return truth value of whether the initialization operation was successful. 
22        * @see ChameleonDeviceConfig 
23        */ 
24       boolean chameleonUSBInterfaceInitialize(Activity mainActivity); 
25    
26       /** 
27        * Call one of these functions to initialize the Chameleon USB library. 
28        * (!!) THIS MUST BE CALLED (!!) before performing any operations with this 
29        * interface, or for most of the operations in the ChameleonDeviceConfig class. 
30        * @param mainActivity : a reference to the main Android app's activity (a simple "this" 
31        *                       reference should do it if called from within an activity method). 
32        * @param libraryLoggingLevel : level of detail in logging events and data generated by the library. 
33        * @return truth value of whether the initialization operation was successful. 
34        * @see ChameleonDeviceConfig 
35        */ 
36       boolean chameleonUSBInterfaceInitialize(Activity mainActivity, LocalLoggingLevel libraryLoggingLevel); 
37    
38       /** 
39        * Call this method to close all Chameleon device connections and shutdown any running 
40        * serial USB routines and/or callbacks. This function will also shutdown the 
41        * active foreground service used to listen for new Chameleon device connections over the 
42        * Android device's USB input. DO NOT RE-INITIALIZE THIS LIBRARY AGAIN WITHOUT FIRST 
43        * CALLING THIS FUNCTION TO DE-INITIALIZE THE EXISTING CONFIGURED INSTANCE!! 
44        * @return truth value of whether the operation was successful. 
45        */ 
46       boolean chameleonUSBInterfaceShutdown(); 
47    
48       /** 
49        * Tests for whether a Chameleon Mini device is attached to the Android device over USB and 
50        * appears to be operating correctly. 
51        * @return true if a working Chameleon device is attached to the Android phone. 
52        */ 
53       boolean chameleonPresent(); 
54    
55       /** 
56        * Tests for whether a Chameleon Mini device is attached to the Android device over USB and 
57        * appears to be operating correctly. 
58        * @param expectedRevType : The expected revision of the connected Chameleon Mini board. 
59        * @return true if a working Chameleon device of the specified type is attached to the Android phone. 
60        * @ref ChameleonDeviceConfig.isRevisionEDevice() 
61        * @ref ChameleonDeviceConfig.isRevisionGDevice() 
62        */ 
63       boolean chameleonPresent(ChameleonBoardType_t expectedRevType); 
64    
65       /** 
66        * Select and optionally prepare slot 1-8 on the Chameleon board for an image to be 
67        * uploaded via XModem. Note that if a slot corresponds to a different configuration 
68        * (say, for example, a MIFARE1K vs. MIFARE4K slot, or configs with differing UID sizes), 
69        * then it is possible in my experience that an upload of a card image over XModem may fail 
70        * to correctly update the contents of the current slot. Therefore, it is my 
71        * recommendation that before any XModem upload attempt of an ingested card dump: 
72        * 1) The desired slot number is selected; 2) The existing contents are cleared from that 
73        * slot; and 3) The configuration type is preset to the type of image that will be 
74        * uploaded into the Chameleon over XModem. These two functions will help prepare for the 
75        * ingested Mifare dumps to be sucessfully uploaded consistently without wierd and/or 
76        * semi-unpredictable behaviors... 
77        * @param slotNumber 
78        * @param clearSlot 
79        * @return 
80        */ 
81       boolean prepareChameleonEmulationSlot(@IntRange(from=1,to=8) int slotNumber, boolean clearSlot); 
82       boolean prepareChameleonEmulationSlot(@IntRange(from=1,to=8) int slotNumber, boolean clearSlot, 
83                                             ChameleonEmulatedConfigType_t chameleonConfigType); 
84    
85       /** 
86        * Initiates an upload of the ingested dump to the current Chameleon slot via XModem. 
87        * Notice that we have amended the proposed library spec here so that the configuration of the 
88        * card type, slot to be written into, and the pre-clearing of this slot are handled by a 
89        * separate routine to be called before these upload dump functions. 
90        * @param tagDataBytes : the ingested data bytes (a.k.a. Mifare dumps) to be written to the Chameleon. 
91        * @return The new UID of the emulated tag after the upload. 
92        * @ref prepareChameleonEmulationSlot 
93        */ 
94       String chameleonUpload(byte[] tagDataBytes); 
95       String chameleonUpload(InputStream dumpDataStream);
